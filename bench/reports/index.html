🚀 Quick Start

Get running in seconds with Docker

# One-line install (Linux/macOS/Windows)
curl -fsSL https://raw.githubusercontent.com/qades/apimap/main/scripts/install.sh | bash

# Start the server
apimap start

# Or use Docker Compose
docker-compose up -d
# config/config.yaml
providers:
  openai:
    apiKeyEnv: "OPENAI_API_KEY"
  anthropic:
    apiKeyEnv: "ANTHROPIC_API_KEY"
  ollama:
    baseUrl: "http://localhost:11434"

routes:
  # Route Claude requests to Anthropic
  - pattern: "claude-*"
    provider: anthropic
    priority: 100
  
  # Route GPT requests to OpenAI
  - pattern: "gpt-*"
    provider: openai
    priority: 90
  
  # Route local/* to Ollama
  - pattern: "local/*"
    provider: ollama
    model: "${1}"
    priority: 80
# Use any OpenAI-compatible client
curl http://localhost:3000/v1/chat/completions \
  -H "Authorization: Bearer your-key" \
  -d '{
    "model": "claude-3-opus",
    "messages": [{"role": "user", "content": "Hello!"}]
  }'

# Or Anthropic format
curl http://localhost:3000/v1/messages \
  -H "x-api-key: your-key" \
  -d '{
    "model": "gpt-4o",
    "messages": [{"role": "user", "content": "Hello!"}]
  }'
